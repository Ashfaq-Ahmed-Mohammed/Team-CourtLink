SQL Queries

1. Customer Table

CREATE TABLE Customer (
    Customer_ID INTEGER PRIMARY KEY
                        UNIQUE
                        NOT NULL,
    Name        TEXT    NOT NULL,
    Contact     TEXT    UNIQUE
                        NOT NULL,
    Email       TEXT    NOT NULL
);

2. Sport Table

CREATE TABLE Sport (
    Sport_ID          INTEGER PRIMARY KEY
                              UNIQUE
                              NOT NULL,
    Sport_name        TEXT    UNIQUE
                              NOT NULL,
    Sport_Description TEXT
);
